132 def execute_webhook(self, *, payload, wait=False, file=None, files=None): 

133 cleanup = None 

134 if file is not None: 

135 multipart = { 

136 'file': (file.filename, file.fp, 'application/octet-stream'), 

137 'payload_json': utils.to_json(payload) 

138 } 

139 data = None 

140 cleanup = file.close 

141 files_to_pass = [file] 

142 elif files is not None: 

143 multipart = { 

144 'payload_json': utils.to_json(payload) 

145 } 

146 for i, file in enumerate(files): 

147 multipart['file%i' % i] = (file.filename, file.fp, 'application/octet-stream') 

148 data = None 

149 

150 def _anon(): 

151 for f in files: 

152 f.close() 

153 

154 cleanup = _anon 

155 files_to_pass = files 

156 else: 

157 data = payload 

158 multipart = None 

159 files_to_pass = None 

160 

161 url = '%s?wait=%d' % (self._request_url, wait) 

162 maybe_coro = None 

163 try: 

164 maybe_coro = self.request('POST', url, multipart=multipart, payload=data, files=files_to_pass) 

165 finally: 

166 if maybe_coro is not None and cleanup is not None: 

167 if not asyncio.iscoroutine(maybe_coro): 

168 cleanup() 

169 else: 

170 maybe_coro = self._wrap_coroutine_and_cleanup(maybe_coro, cleanup) 

171 

172 # if request raises up there then this should never be `None` 

173 return self.handle_execution_response(maybe_coro, wait=wait)

175class AsyncWebhookAdapter(WebhookAdapter): 

176 """A webhook adapter suited for use with aiohttp. 

177 

178 .. note:: 

179 

180 You are responsible for cleaning up the client session. 

181 

182 Parameters 

183 ----------- 

184 session: :class:`aiohttp.ClientSession` 

185 The session to use to send requests. 

186 """ 

187 

188 def __init__(self, session): 

189 self.session = session 

190 self.loop = asyncio.get_event_loop() 

191 

192 def is_async(self): 

193 return True 

194 

195 async def request(self, verb, url, payload=None, multipart=None, *, files=None, reason=None): 

196 headers = {} 

197 data = None 

198 files = files or [] 

199 if payload: 

200 headers['Content-Type'] = 'application/json' 

201 data = utils.to_json(payload) 

202 

203 if reason: 

204 headers['X-Audit-Log-Reason'] = _uriquote(reason, safe='/ ') 

205 

206 

207 base_url = url.replace(self._request_url, '/') or '/' 

208 _id = self._webhook_id 

209 for tries in range(5): 

210 for file in files: 

211 file.reset(seek=tries) 

212 

213 if multipart: 

214 data = aiohttp.FormData() 

215 for key, value in multipart.items(): 

216 if key.startswith('file'): 

217 data.add_field(key, value[1], filename=value[0], content_type=value[2]) 

218 else: 

219 data.add_field(key, value) 

220 

221 async with self.session.request(verb, url, headers=headers, data=data) as r: 

222 log.debug('Webhook ID %s with %s %s has returned status code %s', _id, verb, base_url, r.status) 

223 # Coerce empty strings to return None for hygiene purposes 

224 response = (await r.text(encoding='utf-8')) or None 

225 if r.headers['Content-Type'] == 'application/json': 

226 response = json.loads(response) 

227 

228 # check if we have rate limit header information 

229 remaining = r.headers.get('X-Ratelimit-Remaining') 

230 if remaining == '0' and r.status != 429: 

231 delta = utils._parse_ratelimit_header(r) 

232 log.debug('Webhook ID %s has been pre-emptively rate limited, waiting %.2f seconds', _id, delta) 

233 await asyncio.sleep(delta) 

234 

235 if 300 > r.status >= 200: 

236 return response 

237 

238 # we are being rate limited 

239 if r.status == 429: 

240 if not r.headers.get('Via'): 

241 # Banned by Cloudflare more than likely. 

242 raise HTTPException(r, data) 

243 

244 retry_after = response['retry_after'] / 1000.0 

245 log.warning('Webhook ID %s is rate limited. Retrying in %.2f seconds', _id, retry_after) 

246 await asyncio.sleep(retry_after) 

247 continue 

248 

249 if r.status in (500, 502): 

250 await asyncio.sleep(1 + tries * 2) 

251 continue 

252 

253 if r.status == 403: 

254 raise Forbidden(r, response) 

255 elif r.status == 404: 

256 raise NotFound(r, response) 

257 else: 

258 raise HTTPException(r, response) 

259 

260 # no more retries 

261 if r.status >= 500: 

262 raise DiscordServerError(r, response) 

263 raise HTTPException(r, response) 

264 

265 async def handle_execution_response(self, response, *, wait): 

266 data = await response 

267 if not wait: 

268 return data 

269 

270 # transform into Message object 

271 # Make sure to coerce the state to the partial one to allow message edits/delete 

272 state = _PartialWebhookState(self, self.webhook, parent=self.webhook._state) 

273 return WebhookMessage(data=data, state=state, channel=self.webhook.channel)

519class Webhook(Hashable): 

520 """Represents a Discord webhook. 

521 

522 Webhooks are a form to send messages to channels in Discord without a 

523 bot user or authentication. 

524 

525 There are two main ways to use Webhooks. The first is through the ones 

526 received by the library such as :meth:`.Guild.webhooks` and 

527 :meth:`.TextChannel.webhooks`. The ones received by the library will 

528 automatically have an adapter bound using the library's HTTP session. 

529 Those webhooks will have :meth:`~.Webhook.send`, :meth:`~.Webhook.delete` and 

530 :meth:`~.Webhook.edit` as coroutines. 

531 

532 The second form involves creating a webhook object manually without having 

533 it bound to a websocket connection using the :meth:`~.Webhook.from_url` or 

534 :meth:`~.Webhook.partial` classmethods. This form allows finer grained control 

535 over how requests are done, allowing you to mix async and sync code using either 

536 :doc:`aiohttp <aio:index>` or :doc:`req:index`. 

537 

538 For example, creating a webhook from a URL and using :doc:`aiohttp <aio:index>`: 

539 

540 .. code-block:: python3 

541 

542 from discord import Webhook, AsyncWebhookAdapter 

543 import aiohttp 

544 

545 async def foo(): 

546 async with aiohttp.ClientSession() as session: 

547 webhook = Webhook.from_url('url-here', adapter=AsyncWebhookAdapter(session)) 

548 await webhook.send('Hello World', username='Foo') 

549 

550 Or creating a webhook from an ID and token and using :doc:`req:index`: 

551 

552 .. code-block:: python3 

553 

554 import requests 

555 from discord import Webhook, RequestsWebhookAdapter 

556 

557 webhook = Webhook.partial(123456, 'abcdefg', adapter=RequestsWebhookAdapter()) 

558 webhook.send('Hello World', username='Foo') 

559 

560 .. container:: operations 

561 

562 .. describe:: x == y 

563 

564 Checks if two webhooks are equal. 

565 

566 .. describe:: x != y 

567 

568 Checks if two webhooks are not equal. 

569 

570 .. describe:: hash(x) 

571 

572 Returns the webhooks's hash. 

573 

574 .. versionchanged:: 1.4 

575 Webhooks are now comparable and hashable. 

576 

577 Attributes 

578 ------------ 

579 id: :class:`int` 

580 The webhook's ID 

581 type: :class:`WebhookType` 

582 The type of the webhook. 

583 

584 .. versionadded:: 1.3 

585 

586 token: Optional[:class:`str`] 

587 The authentication token of the webhook. If this is ``None`` 

588 then the webhook cannot be used to make requests. 

589 guild_id: Optional[:class:`int`] 

590 The guild ID this webhook is for. 

591 channel_id: Optional[:class:`int`] 

592 The channel ID this webhook is for. 

593 user: Optional[:class:`abc.User`] 

594 The user this webhook was created by. If the webhook was 

595 received without authentication then this will be ``None``. 

596 name: Optional[:class:`str`] 

597 The default name of the webhook. 

598 avatar: Optional[:class:`str`] 

599 The default avatar of the webhook. 

600 """

757 def avatar_url_as(self, *, format=None, size=1024): 

758 """Returns an :class:`Asset` for the avatar the webhook has. 

759 

760 If the webhook does not have a traditional avatar, an asset for 

761 the default avatar is returned instead. 

762 

763 The format must be one of 'jpeg', 'jpg', or 'png'. 

764 The size must be a power of 2 between 16 and 1024. 

765 

766 Parameters 

767 ----------- 

768 format: Optional[:class:`str`] 

769 The format to attempt to convert the avatar to. 

770 If the format is ``None``, then it is equivalent to png. 

771 size: :class:`int` 

772 The size of the image to display. 

773 

774 Raises 

775 ------ 

776 InvalidArgument 

777 Bad image format passed to ``format`` or invalid ``size``. 

778 

779 Returns 

780 -------- 

781 :class:`Asset` 

782 The resulting CDN asset. 

783 """ 

784 if self.avatar is None: 

785 # Default is always blurple apparently 

786 return Asset(self._state, '/embed/avatars/0.png') 

787 

788 if not utils.valid_icon_size(size): 

789 raise InvalidArgument("size must be a power of 2 between 16 and 1024") 

790 

791 format = format or 'png' 

792 

793 if format not in ('png', 'jpg', 'jpeg'): 

794 raise InvalidArgument("format must be one of 'png', 'jpg', or 'jpeg'.") 

795 

796 url = '/avatars/{0.id}/{0.avatar}.{1}?size={2}'.format(self, format, size) 

797 return Asset(self._state, url)

885 def send(self, content=None, *, wait=False, username=None, avatar_url=None, tts=False, 

886 file=None, files=None, embed=None, embeds=None, allowed_mentions=None): 

887 """|maybecoro| 

888 

889 Sends a message using the webhook. 

890 

891 If the webhook is constructed with a :class:`RequestsWebhookAdapter` then this is 

892 not a coroutine. 

893 

894 The content must be a type that can convert to a string through ``str(content)``. 

895 

896 To upload a single file, the ``file`` parameter should be used with a 

897 single :class:`File` object. 

898 

899 If the ``embed`` parameter is provided, it must be of type :class:`Embed` and 

900 it must be a rich embed type. You cannot mix the ``embed`` parameter with the 

901 ``embeds`` parameter, which must be a :class:`list` of :class:`Embed` objects to send. 

902 

903 Parameters 

904 ------------ 

905 content: :class:`str` 

906 The content of the message to send. 

907 wait: :class:`bool` 

908 Whether the server should wait before sending a response. This essentially 

909 means that the return type of this function changes from ``None`` to 

910 a :class:`WebhookMessage` if set to ``True``. 

911 username: :class:`str` 

912 The username to send with this message. If no username is provided 

913 then the default username for the webhook is used. 

914 avatar_url: Union[:class:`str`, :class:`Asset`] 

915 The avatar URL to send with this message. If no avatar URL is provided 

916 then the default avatar for the webhook is used. 

917 tts: :class:`bool` 

918 Indicates if the message should be sent using text-to-speech. 

919 file: :class:`File` 

920 The file to upload. This cannot be mixed with ``files`` parameter. 

921 files: List[:class:`File`] 

922 A list of files to send with the content. This cannot be mixed with the 

923 ``file`` parameter. 

924 embed: :class:`Embed` 

925 The rich embed for the content to send. This cannot be mixed with 

926 ``embeds`` parameter. 

927 embeds: List[:class:`Embed`] 

928 A list of embeds to send with the content. Maximum of 10. This cannot 

929 be mixed with the ``embed`` parameter. 

930 allowed_mentions: :class:`AllowedMentions` 

931 Controls the mentions being processed in this message. 

932 

933 .. versionadded:: 1.4 

934 

935 Raises 

936 -------- 

937 HTTPException 

938 Sending the message failed. 

939 NotFound 

940 This webhook was not found. 

941 Forbidden 

942 The authorization token for the webhook is incorrect. 

943 InvalidArgument 

944 You specified both ``embed`` and ``embeds`` or the length of 

945 ``embeds`` was invalid or there was no token associated with 

946 this webhook. 

947 

948 Returns 

949 --------- 

950 Optional[:class:`WebhookMessage`] 

951 The message that was sent. 

952 """ 

953 

954 payload = {} 

955 if self.token is None: 

956 raise InvalidArgument('This webhook does not have a token associated with it') 

957 if files is not None and file is not None: 

958 raise InvalidArgument('Cannot mix file and files keyword arguments.') 

959 if embeds is not None and embed is not None: 

960 raise InvalidArgument('Cannot mix embed and embeds keyword arguments.') 

961 

962 if embeds is not None: 

963 if len(embeds) > 10: 

964 raise InvalidArgument('embeds has a maximum of 10 elements.') 

965 payload['embeds'] = [e.to_dict() for e in embeds] 

966 

967 if embed is not None: 

968 payload['embeds'] = [embed.to_dict()] 

969 

970 if content is not None: 

971 payload['content'] = str(content) 

972 

973 payload['tts'] = tts 

974 if avatar_url: 

975 payload['avatar_url'] = str(avatar_url) 

976 if username: 

977 payload['username'] = username 

978 

979 previous_mentions = getattr(self._state, 'allowed_mentions', None) 

980 

981 if allowed_mentions: 

982 if previous_mentions is not None: 

983 payload['allowed_mentions'] = previous_mentions.merge(allowed_mentions).to_dict() 

984 else: 

985 payload['allowed_mentions'] = allowed_mentions.to_dict() 

986 elif previous_mentions is not None: 

987 payload['allowed_mentions'] = previous_mentions.to_dict() 

988 

989 return self._adapter.execute_webhook(wait=wait, file=file, files=files, payload=payload)

995 def edit_message(self, message_id, **fields): 

996 """|maybecoro| 

997 

998 Edits a message owned by this webhook. 

999 

1000 This is a lower level interface to :meth:`WebhookMessage.edit` in case 

1001 you only have an ID. 

1002 

1003 .. versionadded:: 1.6 

1004 

1005 Parameters 

1006 ------------ 

1007 message_id: :class:`int` 

1008 The message ID to edit. 

1009 content: Optional[:class:`str`] 

1010 The content to edit the message with or ``None`` to clear it. 

1011 embeds: List[:class:`Embed`] 

1012 A list of embeds to edit the message with. 

1013 embed: Optional[:class:`Embed`] 

1014 The embed to edit the message with. ``None`` suppresses the embeds. 

1015 This should not be mixed with the ``embeds`` parameter. 

1016 allowed_mentions: :class:`AllowedMentions` 

1017 Controls the mentions being processed in this message. 

1018 See :meth:`.abc.Messageable.send` for more information. 

1019 

1020 Raises 

1021 ------- 

1022 HTTPException 

1023 Editing the message failed. 

1024 Forbidden 

1025 Edited a message that is not yours. 

1026 InvalidArgument 

1027 You specified both ``embed`` and ``embeds`` or the length of 

1028 ``embeds`` was invalid or there was no token associated with 

1029 this webhook. 

1030 """ 

1031 

1032 payload = {} 

1033 

1034 if self.token is None: 

1035 raise InvalidArgument('This webhook does not have a token associated with it') 

1036 

1037 try: 

1038 content = fields['content'] 

1039 except KeyError: 

1040 pass 

1041 else: 

1042 if content is not None: 

1043 content = str(content) 

1044 payload['content'] = content 

1045 

1046 # Check if the embeds interface is being used 

1047 try: 

1048 embeds = fields['embeds'] 

1049 except KeyError: 

1050 # Nope 

1051 pass 

1052 else: 

1053 if embeds is None or len(embeds) > 10: 

1054 raise InvalidArgument('embeds has a maximum of 10 elements') 

1055 payload['embeds'] = [e.to_dict() for e in embeds] 

1056 

1057 try: 

1058 embed = fields['embed'] 

1059 except KeyError: 

1060 pass 

1061 else: 

1062 if 'embeds' in payload: 

1063 raise InvalidArgument('Cannot mix embed and embeds keyword arguments') 

1064 

1065 if embed is None: 

1066 payload['embeds'] = [] 

1067 else: 

1068 payload['embeds'] = [embed.to_dict()] 

1069 

1070 allowed_mentions = fields.pop('allowed_mentions', None) 

1071 previous_mentions = getattr(self._state, 'allowed_mentions', None) 

1072 

1073 if allowed_mentions: 

1074 if previous_mentions is not None: 

1075 payload['allowed_mentions'] = previous_mentions.merge(allowed_mentions).to_dict() 

1076 else: 

1077 payload['allowed_mentions'] = allowed_mentions.to_dict() 

1078 elif previous_mentions is not None: 

1079 payload['allowed_mentions'] = previous_mentions.to_dict() 

1080 

1081 return self._adapter.edit_webhook_message(message_id, payload=payload)
