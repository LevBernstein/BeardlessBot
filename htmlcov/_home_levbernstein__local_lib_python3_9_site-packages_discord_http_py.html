134 async def request(self, route, *, files=None, form=None, **kwargs): 

135 bucket = route.bucket 

136 method = route.method 

137 url = route.url 

138 

139 lock = self._locks.get(bucket) 

140 if lock is None: 

141 lock = asyncio.Lock() 

142 if bucket is not None: 

143 self._locks[bucket] = lock 

144 

145 # header creation 

146 headers = { 

147 'User-Agent': self.user_agent, 

148 'X-Ratelimit-Precision': 'millisecond', 

149 } 

150 

151 if self.token is not None: 

152 headers['Authorization'] = 'Bot ' + self.token if self.bot_token else self.token 

153 # some checking if it's a JSON request 

154 if 'json' in kwargs: 

155 headers['Content-Type'] = 'application/json' 

156 kwargs['data'] = utils.to_json(kwargs.pop('json')) 

157 

158 try: 

159 reason = kwargs.pop('reason') 

160 except KeyError: 

161 pass 

162 else: 

163 if reason: 

164 headers['X-Audit-Log-Reason'] = _uriquote(reason, safe='/ ') 

165 

166 kwargs['headers'] = headers 

167 

168 # Proxy support 

169 if self.proxy is not None: 

170 kwargs['proxy'] = self.proxy 

171 if self.proxy_auth is not None: 

172 kwargs['proxy_auth'] = self.proxy_auth 

173 

174 if not self._global_over.is_set(): 

175 # wait until the global lock is complete 

176 await self._global_over.wait() 

177 

178 await lock.acquire() 

179 with MaybeUnlock(lock) as maybe_lock: 

180 for tries in range(5): 

181 if files: 

182 for f in files: 

183 f.reset(seek=tries) 

184 

185 if form: 

186 form_data = aiohttp.FormData() 

187 for params in form: 

188 form_data.add_field(**params) 

189 kwargs['data'] = form_data 

190 

191 try: 

192 async with self.__session.request(method, url, **kwargs) as r: 

193 log.debug('%s %s with %s has returned %s', method, url, kwargs.get('data'), r.status) 

194 

195 # even errors have text involved in them so this is safe to call 

196 data = await json_or_text(r) 

197 

198 # check if we have rate limit header information 

199 remaining = r.headers.get('X-Ratelimit-Remaining') 

200 if remaining == '0' and r.status != 429: 

201 # we've depleted our current bucket 

202 delta = utils._parse_ratelimit_header(r, use_clock=self.use_clock) 

203 log.debug('A rate limit bucket has been exhausted (bucket: %s, retry: %s).', bucket, delta) 

204 maybe_lock.defer() 

205 self.loop.call_later(delta, lock.release) 

206 

207 # the request was successful so just return the text/json 

208 if 300 > r.status >= 200: 

209 log.debug('%s %s has received %s', method, url, data) 

210 return data 

211 

212 # we are being rate limited 

213 if r.status == 429: 

214 if not r.headers.get('Via'): 

215 # Banned by Cloudflare more than likely. 

216 raise HTTPException(r, data) 

217 

218 fmt = 'We are being rate limited. Retrying in %.2f seconds. Handled under the bucket "%s"' 

219 

220 # sleep a bit 

221 retry_after = data['retry_after'] / 1000.0 

222 log.warning(fmt, retry_after, bucket) 

223 

224 # check if it's a global rate limit 

225 is_global = data.get('global', False) 

226 if is_global: 

227 log.warning('Global rate limit has been hit. Retrying in %.2f seconds.', retry_after) 

228 self._global_over.clear() 

229 

230 await asyncio.sleep(retry_after) 

231 log.debug('Done sleeping for the rate limit. Retrying...') 

232 

233 # release the global lock now that the 

234 # global rate limit has passed 

235 if is_global: 

236 self._global_over.set() 

237 log.debug('Global rate limit is now over.') 

238 

239 continue 

240 

241 # we've received a 500 or 502, unconditional retry 

242 if r.status in {500, 502}: 

243 await asyncio.sleep(1 + tries * 2) 

244 continue 

245 

246 # the usual error cases 

247 if r.status == 403: 

248 raise Forbidden(r, data) 

249 elif r.status == 404: 

250 raise NotFound(r, data) 

251 elif r.status == 503: 

252 raise DiscordServerError(r, data) 

253 else: 

254 raise HTTPException(r, data) 

255 

256 # This is handling exceptions from the request 

257 except OSError as e: 

258 # Connection reset by peer 

259 if tries < 4 and e.errno in (54, 10054): 

260 continue 

261 raise 

262 

263 # We've run out of retries, raise. 

264 if r.status >= 500: 

265 raise DiscordServerError(r, data) 

266 

267 raise HTTPException(r, data)
