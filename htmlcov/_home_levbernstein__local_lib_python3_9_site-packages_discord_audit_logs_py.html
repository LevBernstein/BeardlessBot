100 TRANSFORMERS = { 

101 'verification_level': (None, _transform_verification_level), 

102 'explicit_content_filter': (None, _transform_explicit_content_filter), 

103 'allow': (None, _transform_permissions), 

104 'deny': (None, _transform_permissions),

116 'icon_hash': ('icon', None),

122 def __init__(self, entry, data): 

123 self.before = AuditLogDiff() 

124 self.after = AuditLogDiff() 

125 

126 for elem in data: 

127 attr = elem['key'] 

128 

129 # special cases for role add/remove 

130 if attr == '$add': 

131 self._handle_role(self.before, self.after, entry, elem['new_value']) 

132 continue 

133 elif attr == '$remove': 

134 self._handle_role(self.after, self.before, entry, elem['new_value']) 

135 continue 

136 

137 transformer = self.TRANSFORMERS.get(attr) 

138 if transformer: 

139 key, transformer = transformer 

140 if key: 

141 attr = key 

142 

143 try: 

144 before = elem['old_value'] 

145 except KeyError: 

146 before = None 

147 else: 

148 if transformer: 

149 before = transformer(entry, before) 

150 

151 setattr(self.before, attr, before) 

152 

153 try: 

154 after = elem['new_value'] 

155 except KeyError: 

156 after = None 

157 else: 

158 if transformer: 

159 after = transformer(entry, after) 

160 

161 setattr(self.after, attr, after) 

162 

163 # add an alias 

164 if hasattr(self.after, 'colour'): 

165 self.after.color = self.after.colour 

166 self.before.color = self.before.colour 

167 

168 def __repr__(self): 

169 return '<AuditLogChanges before=%r after=%r>' % (self.before, self.after)

171 def _handle_role(self, first, second, entry, elem): 

172 if not hasattr(first, 'roles'): 

173 setattr(first, 'roles', []) 

174 

175 data = [] 

176 g = entry.guild 

177 

178 for e in elem: 

179 role_id = int(e['id']) 

180 role = g.get_role(role_id) 

181 

182 if role is None: 

183 role = Object(id=role_id) 

184 role.name = e['name'] 

185 

186 data.append(role) 

187 

188 setattr(second, 'roles', data)

190class AuditLogEntry(Hashable): 

191 r"""Represents an Audit Log entry. 

192 

193 You retrieve these via :meth:`Guild.audit_logs`. 

194 

195 .. container:: operations 

196 

197 .. describe:: x == y 

198 

199 Checks if two entries are equal. 

200 

201 .. describe:: x != y 

202 

203 Checks if two entries are not equal. 

204 

205 .. describe:: hash(x) 

206 

207 Returns the entry's hash. 

208 

209 .. versionchanged:: 1.7 

210 Audit log entries are now comparable and hashable. 

211 

212 Attributes 

213 ----------- 

214 action: :class:`AuditLogAction` 

215 The action that was done. 

216 user: :class:`abc.User` 

217 The user who initiated this action. Usually a :class:`Member`\, unless gone 

218 then it's a :class:`User`. 

219 id: :class:`int` 

220 The entry ID. 

221 target: Any 

222 The target that got changed. The exact type of this depends on 

223 the action being done. 

224 reason: Optional[:class:`str`] 

225 The reason this action was done. 

226 extra: Any 

227 Extra information that this entry has that might be useful. 

228 For most actions, this is ``None``. However in some cases it 

229 contains extra information. See :class:`AuditLogAction` for 

230 which actions have this field filled out. 

231 """
