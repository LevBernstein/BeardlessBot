66def pytest_terminal_summary(terminalreporter: "TerminalReporter") -> None: 

67 durations = terminalreporter.config.option.durations 

68 durations_min = terminalreporter.config.option.durations_min 

69 verbose = terminalreporter.config.getvalue("verbose") 

70 if durations is None: 

71 return 

72 tr = terminalreporter 

73 dlist = [] 

74 for replist in tr.stats.values(): 

75 for rep in replist: 

76 if hasattr(rep, "duration"): 

77 dlist.append(rep) 

78 if not dlist: 

79 return 

80 dlist.sort(key=lambda x: x.duration, reverse=True) # type: ignore[no-any-return] 

81 if not durations: 

82 tr.write_sep("=", "slowest durations") 

83 else: 

84 tr.write_sep("=", "slowest %s durations" % durations) 

85 dlist = dlist[:durations] 

86 

87 for i, rep in enumerate(dlist): 

88 if verbose < 2 and rep.duration < durations_min: 

89 tr.write_line("") 

90 tr.write_line( 

91 "(%s durations < %gs hidden. Use -vv to show these durations.)" 

92 % (len(dlist) - i, durations_min) 

93 ) 

94 break 

95 tr.write_line(f"{rep.duration:02.2f}s {rep.when:<8} {rep.nodeid}")

114def runtestprotocol( 

115 item: Item, log: bool = True, nextitem: Optional[Item] = None 

116) -> List[TestReport]: 

117 hasrequest = hasattr(item, "_request") 

118 if hasrequest and not item._request: # type: ignore[attr-defined] 

119 item._initrequest() # type: ignore[attr-defined] 

120 rep = call_and_report(item, "setup", log) 

121 reports = [rep] 

122 if rep.passed: 

123 if item.config.getoption("setupshow", False): 

124 show_test_item(item) 

125 if not item.config.getoption("setuponly", False): 

126 reports.append(call_and_report(item, "call", log)) 

127 reports.append(call_and_report(item, "teardown", log, nextitem=nextitem)) 

128 # After all teardown hooks have been called 

129 # want funcargs and request info to go away. 

130 if hasrequest: 

131 item._request = False # type: ignore[attr-defined] 

132 item.funcargs = None # type: ignore[attr-defined] 

133 return reports

225def check_interactive_exception(call: "CallInfo[object]", report: BaseReport) -> bool: 

226 """Check whether the call raised an exception that should be reported as 

227 interactive.""" 

228 if call.excinfo is None: 

229 # Didn't raise. 

230 return False 

231 if hasattr(report, "wasxfail"): 

232 # Exception was expected. 

233 return False

264class CallInfo(Generic[TResult]): 

265 """Result/Exception info a function invocation. 

266 

267 :param T result: 

268 The return value of the call, if it didn't raise. Can only be 

269 accessed if excinfo is None. 

270 :param Optional[ExceptionInfo] excinfo: 

271 The captured exception of the call, if it raised. 

272 :param float start: 

273 The system time when the call started, in seconds since the epoch. 

274 :param float stop: 

275 The system time when the call ended, in seconds since the epoch. 

276 :param float duration: 

277 The call duration, in seconds. 

278 :param str when: 

279 The context of invocation: "setup", "call", "teardown", ... 

280 """ 

281 

282 _result = attr.ib(type="Optional[TResult]") 

283 excinfo = attr.ib(type=Optional[ExceptionInfo[BaseException]]) 

284 start = attr.ib(type=float) 

285 stop = attr.ib(type=float) 

286 duration = attr.ib(type=float) 

287 when = attr.ib(type="Literal['collect', 'setup', 'call', 'teardown']") 

288 

289 @property 

290 def result(self) -> TResult: 

291 if self.excinfo is not None: 

292 raise AttributeError(f"{self!r} has no valid result") 

293 # The cast is safe because an exception wasn't raised, hence 

294 # _result has the expected function return type (which may be 

295 # None, that's why a cast and not an assert). 

296 return cast(TResult, self._result) 

297 

298 @classmethod 

299 def from_call( 

300 cls, 

301 func: "Callable[[], TResult]", 

302 when: "Literal['collect', 'setup', 'call', 'teardown']", 

303 reraise: Optional[ 

304 Union[Type[BaseException], Tuple[Type[BaseException], ...]] 

305 ] = None, 

306 ) -> "CallInfo[TResult]": 

307 excinfo = None 

308 start = timing.time() 

309 precise_start = timing.perf_counter() 

310 try: 

311 result: Optional[TResult] = func() 

312 except BaseException: 

313 excinfo = ExceptionInfo.from_current() 

314 if reraise is not None and isinstance(excinfo.value, reraise): 

315 raise 

316 result = None 

317 # use the perf counter 

318 precise_stop = timing.perf_counter() 

319 duration = precise_stop - precise_start 

320 stop = timing.time() 

321 return cls( 

322 start=start, 

323 stop=stop, 

324 duration=duration, 

325 when=when, 

326 result=result, 

327 excinfo=excinfo, 

328 )

340def pytest_make_collect_report(collector: Collector) -> CollectReport: 

341 call = CallInfo.from_call(lambda: list(collector.collect()), "collect") 

342 longrepr: Union[None, Tuple[str, int, str], str, TerminalRepr] = None 

343 if not call.excinfo: 

344 outcome: Literal["passed", "skipped", "failed"] = "passed" 

345 else: 

346 skip_exceptions = [Skipped] 

347 unittest = sys.modules.get("unittest") 

348 if unittest is not None: 

349 # Type ignored because unittest is loaded dynamically. 

350 skip_exceptions.append(unittest.SkipTest) # type: ignore 

351 if isinstance(call.excinfo.value, tuple(skip_exceptions)): 

352 outcome = "skipped" 

353 r_ = collector._repr_failure_py(call.excinfo, "line") 

354 assert isinstance(r_, ExceptionChainRepr), repr(r_) 

355 r = r_.reprcrash 

356 assert r 

357 longrepr = (str(r.path), r.lineno, r.message) 

358 else: 

359 outcome = "failed" 

360 errorinfo = collector.repr_failure(call.excinfo) 

361 if not hasattr(errorinfo, "toterminal"): 

362 assert isinstance(errorinfo, str) 

363 errorinfo = CollectErrorRepr(errorinfo) 

364 longrepr = errorinfo 

365 result = call.result if not call.excinfo else None 

366 rep = CollectReport(collector.nodeid, outcome, longrepr, result) 

367 rep.call = call # type: ignore # see collect_one_node 

368 return rep

371class SetupState: 

372 """Shared state for setting up/tearing down test items or collectors.""" 

373 

374 def __init__(self): 

375 self.stack: List[Node] = [] 

376 self._finalizers: Dict[Node, List[Callable[[], object]]] = {} 

377 

378 def addfinalizer(self, finalizer: Callable[[], object], colitem) -> None: 

379 """Attach a finalizer to the given colitem.""" 

380 assert colitem and not isinstance(colitem, tuple) 

381 assert callable(finalizer) 

382 # assert colitem in self.stack # some unit tests don't setup stack :/ 

383 self._finalizers.setdefault(colitem, []).append(finalizer) 

384 

385 def _pop_and_teardown(self): 

386 colitem = self.stack.pop() 

387 self._teardown_with_finalization(colitem) 

388 

389 def _callfinalizers(self, colitem) -> None: 

390 finalizers = self._finalizers.pop(colitem, None) 

391 exc = None 

392 while finalizers: 

393 fin = finalizers.pop() 

394 try: 

395 fin() 

396 except TEST_OUTCOME as e: 

397 # XXX Only first exception will be seen by user, 

398 # ideally all should be reported. 

399 if exc is None: 

400 exc = e 

401 if exc: 

402 raise exc 

403 

404 def _teardown_with_finalization(self, colitem) -> None: 

405 self._callfinalizers(colitem) 

406 colitem.teardown() 

407 for colitem in self._finalizers: 

408 assert colitem in self.stack 

409 

410 def teardown_all(self) -> None: 

411 while self.stack: 

412 self._pop_and_teardown() 

413 for key in list(self._finalizers): 

414 self._teardown_with_finalization(key) 

415 assert not self._finalizers 

416 

417 def teardown_exact(self, item, nextitem) -> None: 

418 needed_collectors = nextitem and nextitem.listchain() or [] 

419 self._teardown_towards(needed_collectors) 

420 

421 def _teardown_towards(self, needed_collectors) -> None: 

422 exc = None 

423 while self.stack: 

424 if self.stack == needed_collectors[: len(self.stack)]: 

425 break 

426 try: 

427 self._pop_and_teardown() 

428 except TEST_OUTCOME as e: 

429 # XXX Only first exception will be seen by user, 

430 # ideally all should be reported. 

431 if exc is None: 

432 exc = e 

433 if exc: 

434 raise exc 

435 

436 def prepare(self, colitem) -> None: 

437 """Setup objects along the collector chain to the test-method.""" 

438 

439 # Check if the last collection node has raised an error. 

440 for col in self.stack: 

441 if hasattr(col, "_prepare_exc"): 

442 exc = col._prepare_exc # type: ignore[attr-defined] 

443 raise exc 

444 

445 needed_collectors = colitem.listchain() 

446 for col in needed_collectors[len(self.stack) :]: 

447 self.stack.append(col) 

448 try: 

449 col.setup() 

450 except TEST_OUTCOME as e: 

451 col._prepare_exc = e # type: ignore[attr-defined] 

452 raise e
