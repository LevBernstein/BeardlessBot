56def fill_with_flags(*, inverted=False): 

57 def decorator(cls): 

58 cls.VALID_FLAGS = { 

59 name: value.flag 

60 for name, value in cls.__dict__.items() 

61 if isinstance(value, flag_value) 

62 } 

63 

64 if inverted: 

65 max_bits = max(cls.VALID_FLAGS.values()).bit_length() 

66 cls.DEFAULT_VALUE = -1 + (2 ** max_bits) 

67 else: 

68 cls.DEFAULT_VALUE = 0 

69 

70 return cls 

71 return decorator

82 setattr(self, key, value)

122class SystemChannelFlags(BaseFlags): 

123 r"""Wraps up a Discord system channel flag value. 

124 

125 Similar to :class:`Permissions`\, the properties provided are two way. 

126 You can set and retrieve individual bits using the properties as if they 

127 were regular bools. This allows you to edit the system flags easily. 

128 

129 To construct an object you can pass keyword arguments denoting the flags 

130 to enable or disable. 

131 

132 .. container:: operations 

133 

134 .. describe:: x == y 

135 

136 Checks if two flags are equal. 

137 .. describe:: x != y 

138 

139 Checks if two flags are not equal. 

140 .. describe:: hash(x) 

141 

142 Return the flag's hash. 

143 .. describe:: iter(x) 

144 

145 Returns an iterator of ``(name, value)`` pairs. This allows it 

146 to be, for example, constructed as a dict or a list of pairs. 

147 

148 Attributes 

149 ----------- 

150 value: :class:`int` 

151 The raw value. This value is a bit array field of a 53-bit integer 

152 representing the currently available flags. You should query 

153 flags via the properties rather than using this raw value. 

154 """ 

155 __slots__ = () 

156 

157 # For some reason the flags for system channels are "inverted" 

158 # ergo, if they're set then it means "suppress" (off in the GUI toggle) 

159 # Since this is counter-intuitive from an API perspective and annoying 

160 # these will be inverted automatically 

161 

162 def _has_flag(self, o): 

163 return (self.value & o) != o 

164 

165 def _set_flag(self, o, toggle): 

166 if toggle is True: 

167 self.value &= ~o 

168 elif toggle is False: 

169 self.value |= o 

170 else: 

171 raise TypeError('Value to set for SystemChannelFlags must be a bool.')

786class MemberCacheFlags(BaseFlags): 

787 """Controls the library's cache policy when it comes to members. 

788 

789 This allows for finer grained control over what members are cached. 

790 Note that the bot's own member is always cached. This class is passed 

791 to the ``member_cache_flags`` parameter in :class:`Client`. 

792 

793 Due to a quirk in how Discord works, in order to ensure proper cleanup 

794 of cache resources it is recommended to have :attr:`Intents.members` 

795 enabled. Otherwise the library cannot know when a member leaves a guild and 

796 is thus unable to cleanup after itself. 

797 

798 To construct an object you can pass keyword arguments denoting the flags 

799 to enable or disable. 

800 

801 The default value is all flags enabled. 

802 

803 .. versionadded:: 1.5 

804 

805 .. container:: operations 

806

810 .. describe:: x != y

816 .. describe:: iter(x)

819 to be, for example, constructed as a dict or a list of pairs. 

820

822 -----------

824 The raw value. You should query flags via the properties
