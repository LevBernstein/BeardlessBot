18class Url(namedtuple('Url', url_attrs)): 

19 """ 

20 Datastructure for representing an HTTP URL. Used as a return value for 

21 :func:`parse_url`. Both the scheme and host are normalized as they are 

22 both case-insensitive according to RFC 3986. 

23 """ 

24 __slots__ = ()

26 def __new__(cls, scheme=None, auth=None, host=None, port=None, path=None, 

27 query=None, fragment=None): 

28 if path and not path.startswith('/'): 

29 path = '/' + path 

30 if scheme: 

31 scheme = scheme.lower() 

32 if host and scheme in NORMALIZABLE_SCHEMES: 

33 host = host.lower()

59 @property 

60 def url(self): 

61 """ 

62 Convert self into a url 

63 

64 This function should more or less round-trip with :func:`.parse_url`. The 

65 returned url may not be exactly the same as the url inputted to 

66 :func:`.parse_url`, but it should be equivalent by the RFC (e.g., urls 

67 with a blank port will have : removed). 

68 

69 Example: :: 

70 

71 >>> U = parse_url('http://google.com/mail/') 

72 >>> U.url 

73 'http://google.com/mail/' 

74 >>> Url('http', 'username:password', 'host.com', 80, 

75 ... '/path', 'query', 'fragment').url 

76 'http://username:password@host.com:80/path?query#fragment' 

77 """ 

78 scheme, auth, host, port, path, query, fragment = self 

79 url = '' 

80 

81 # We use "is not None" we want things to happen with empty strings (or 0 port) 

82 if scheme is not None: 

83 url += scheme + '://' 

84 if auth is not None: 

85 url += auth + '@' 

86 if host is not None: 

87 url += host 

88 if port is not None: 

89 url += ':' + str(port) 

90 if path is not None: 

91 url += path 

92 if query is not None: 

93 url += '?' + query 

94 if fragment is not None: 

95 url += '#' + fragment 

96 

97 return url

103def split_first(s, delims): 

104 """ 

105 Given a string and an iterable of delimiters, split on the first found 

106 delimiter. Return two split parts and the matched delimiter. 

107 

108 If not found, then the first part is the full input string. 

109 

110 Example:: 

111 

112 >>> split_first('foo/bar?baz', '?/=') 

113 ('foo', 'bar?baz', '/') 

114 >>> split_first('foo/bar?baz', '123') 

115 ('foo/bar?baz', '', None) 

116 

117 Scales linearly with number of delims. Not ideal for large number of delims. 

118 """ 

119 min_idx = None 

120 min_delim = None 

121 for d in delims: 

122 idx = s.find(d) 

123 if idx < 0: 

124 continue 

125 

126 if min_idx is None or idx < min_idx: 

127 min_idx = idx 

128 min_delim = d 

129 

130 if min_idx is None or min_idx < 0: 

131 return s, '', None 

132 

133 return s[:min_idx], s[min_idx + 1:], min_delim

136def parse_url(url): 

137 """ 

138 Given a url, return a parsed :class:`.Url` namedtuple. Best-effort is 

139 performed to parse incomplete urls. Fields not provided will be None. 

140 

141 Partly backwards-compatible with :mod:`urlparse`. 

142 

143 Example:: 

144 

145 >>> parse_url('http://google.com/mail/') 

146 Url(scheme='http', host='google.com', port=None, path='/mail/', ...) 

147 >>> parse_url('google.com:80') 

148 Url(scheme=None, host='google.com', port=80, path=None, ...) 

149 >>> parse_url('/foo?bar') 

150 Url(scheme=None, host=None, port=None, path='/foo', query='bar', ...) 

151 """ 

152 

153 # While this code has overlap with stdlib's urlparse, it is much 

154 # simplified for our needs and less annoying. 

155 # Additionally, this implementations does silly things to be optimal 

156 # on CPython. 

157 

158 if not url: 

159 # Empty 

160 return Url() 

161 

162 # Prevent CVE-2019-9740. 

163 # adapted from https://github.com/python/cpython/pull/12755 

164 url = _contains_disallowed_url_pchar_re.sub(lambda match: quote(match.group()), url) 

165 

166 scheme = None 

167 auth = None 

168 host = None 

169 port = None 

170 path = None 

171 fragment = None 

172 query = None 

173 

174 # Scheme 

175 if '://' in url: 

176 scheme, url = url.split('://', 1) 

177 

178 # Find the earliest Authority Terminator 

179 # (http://tools.ietf.org/html/rfc3986#section-3.2) 

180 url, path_, delim = split_first(url, ['/', '?', '#']) 

181 

182 if delim: 

183 # Reassemble the path 

184 path = delim + path_ 

185 

186 # Auth 

187 if '@' in url: 

188 # Last '@' denotes end of auth part 

189 auth, url = url.rsplit('@', 1) 

190 

191 # IPv6 

192 if url and url[0] == '[': 

193 host, url = url.split(']', 1) 

194 host += ']' 

195 

196 # Port 

197 if ':' in url: 

198 _host, port = url.split(':', 1) 

199 

200 if not host: 

201 host = _host 

202 

203 if port: 

204 # If given, ports must be integers. No whitespace, no plus or 

205 # minus prefixes, no non-integer digits such as ^2 (superscript). 

206 if not port.isdigit(): 

207 raise LocationParseError(url) 

208 try: 

209 port = int(port) 

210 except ValueError: 

211 raise LocationParseError(url) 

212 else: 

213 # Blank ports are cool, too. (rfc3986#section-3.2.3) 

214 port = None 

215 

216 elif not host and url: 

217 host = url 

218 

219 if not path: 

220 return Url(scheme, auth, host, port, path, query, fragment) 

221 

222 # Fragment 

223 if '#' in path: 

224 path, fragment = path.split('#', 1) 

225 

226 # Query 

227 if '?' in path: 

228 path, query = path.split('?', 1) 

229 

230 return Url(scheme, auth, host, port, path, query, fragment)
