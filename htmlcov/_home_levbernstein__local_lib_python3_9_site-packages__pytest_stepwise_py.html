65 def pytest_collection_modifyitems( 

66 self, config: Config, items: List[nodes.Item] 

67 ) -> None: 

68 if not self.lastfailed: 

69 self.report_status = "no previously failed tests, not skipping." 

70 return 

71 

72 # check all item nodes until we find a match on last failed 

73 failed_index = None 

74 for index, item in enumerate(items): 

75 if item.nodeid == self.lastfailed: 

76 failed_index = index 

77 break 

78 

79 # If the previously failed test was not found among the test items, 

80 # do not skip any tests. 

81 if failed_index is None: 

82 self.report_status = "previously failed test not found, not skipping." 

83 else: 

84 self.report_status = f"skipping {failed_index} already passed items." 

85 deselected = items[:failed_index]
