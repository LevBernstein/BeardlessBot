38class OggPage:

41 def __init__(self, stream): 

42 try: 

43 header = stream.read(struct.calcsize(self._header.format)) 

44 

45 self.flag, self.gran_pos, self.serial, \ 

46 self.pagenum, self.crc, self.segnum = self._header.unpack(header) 

47 

48 self.segtable = stream.read(self.segnum) 

49 bodylen = sum(struct.unpack('B'*self.segnum, self.segtable)) 

50 self.data = stream.read(bodylen) 

51 except Exception: 

52 raise OggError('bad data stream') from None

54 def iter_packets(self): 

55 packetlen = offset = 0 

56 partial = True 

57 

58 for seg in self.segtable: 

59 if seg == 255: 

60 packetlen += 255 

61 partial = True 

62 else: 

63 packetlen += seg 

64 yield self.data[offset:offset+packetlen], True 

65 offset += packetlen 

66 packetlen = 0
