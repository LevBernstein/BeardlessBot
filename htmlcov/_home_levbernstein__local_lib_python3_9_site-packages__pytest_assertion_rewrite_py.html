360def _read_pyc( 

361 source: Path, pyc: Path, trace: Callable[[str], None] = lambda x: None 

362) -> Optional[types.CodeType]: 

363 """Possibly read a pytest pyc containing rewritten code. 

364 

365 Return rewritten code if successful or None if not. 

366 """ 

367 try: 

368 fp = open(os.fspath(pyc), "rb") 

369 except OSError: 

370 return None 

371 with fp: 

372 # https://www.python.org/dev/peps/pep-0552/ 

373 has_flags = sys.version_info >= (3, 7) 

374 try: 

375 stat_result = os.stat(os.fspath(source)) 

376 mtime = int(stat_result.st_mtime) 

377 size = stat_result.st_size 

378 data = fp.read(16 if has_flags else 12) 

379 except OSError as e: 

380 trace(f"_read_pyc({source}): OSError {e}") 

381 return None 

382 # Check for invalid or out of date pyc file. 

383 if len(data) != (16 if has_flags else 12): 

384 trace("_read_pyc(%s): invalid pyc (too short)" % source) 

385 return None 

386 if data[:4] != importlib.util.MAGIC_NUMBER: 

387 trace("_read_pyc(%s): invalid pyc (bad magic number)" % source) 

388 return None 

389 if has_flags and data[4:8] != b"\x00\x00\x00\x00": 

390 trace("_read_pyc(%s): invalid pyc (unsupported flags)" % source) 

391 return None 

392 mtime_data = data[8 if has_flags else 4 : 12 if has_flags else 8] 

393 if int.from_bytes(mtime_data, "little") != mtime & 0xFFFFFFFF: 

394 trace("_read_pyc(%s): out of date" % source) 

395 return None 

396 size_data = data[12 if has_flags else 8 : 16 if has_flags else 12] 

397 if int.from_bytes(size_data, "little") != size & 0xFFFFFFFF: 

398 trace("_read_pyc(%s): invalid pyc (incorrect size)" % source) 

399 return None 

400 try: 

401 co = marshal.load(fp) 

402 except Exception as e: 

403 trace(f"_read_pyc({source}): marshal.load error {e}") 

404 return None 

405 if not isinstance(co, types.CodeType): 

406 trace("_read_pyc(%s): not a code object" % source) 

407 return None 

408 return co

546def _get_assertion_exprs(src: bytes) -> Dict[int, str]: 

547 """Return a mapping from {lineno: "assertion test expression"}.""" 

548 ret: Dict[int, str] = {} 

549 

550 depth = 0 

551 lines: List[str] = [] 

552 assert_lineno: Optional[int] = None 

553 seen_lines: Set[int] = set() 

554 

555 def _write_and_reset() -> None: 

556 nonlocal depth, lines, assert_lineno, seen_lines 

557 assert assert_lineno is not None 

558 ret[assert_lineno] = "".join(lines).rstrip().rstrip("\\") 

559 depth = 0 

560 lines = [] 

561 assert_lineno = None 

562 seen_lines = set() 

563 

564 tokens = tokenize.tokenize(io.BytesIO(src).readline) 

565 for tp, source, (lineno, offset), _, line in tokens: 

566 if tp == tokenize.NAME and source == "assert": 

567 assert_lineno = lineno 

568 elif assert_lineno is not None: 

569 # keep track of depth for the assert-message `,` lookup 

570 if tp == tokenize.OP and source in "([{": 

571 depth += 1 

572 elif tp == tokenize.OP and source in ")]}": 

573 depth -= 1 

574 

575 if not lines: 

576 lines.append(line[offset:]) 

577 seen_lines.add(lineno) 

578 # a non-nested comma separates the expression from the message 

579 elif depth == 0 and tp == tokenize.OP and source == ",": 

580 # one line assert with message 

581 if lineno in seen_lines and len(lines) == 1: 

582 offset_in_trimmed = offset + len(lines[-1]) - len(line) 

583 lines[-1] = lines[-1][:offset_in_trimmed] 

584 # multi-line assert with message 

585 elif lineno in seen_lines: 

586 lines[-1] = lines[-1][:offset] 

587 # multi line assert with escapd newline before message 

588 else: 

589 lines.append(line[:offset]) 

590 _write_and_reset() 

591 elif tp in {tokenize.NEWLINE, tokenize.ENDMARKER}: 

592 _write_and_reset() 

593 elif lines and lineno not in seen_lines: 

594 lines.append(line) 

595 seen_lines.add(lineno) 

596 

597 return ret

600class AssertionRewriter(ast.NodeVisitor): 

601 """Assertion rewriting implementation. 

602 

603 The main entrypoint is to call .run() with an ast.Module instance, 

604 this will then find all the assert statements and rewrite them to 

605 provide intermediate values and a detailed assertion error. See 

606 http://pybites.blogspot.be/2011/07/behind-scenes-of-pytests-new-assertion.html 

607 for an overview of how this works. 

608 

609 The entry point here is .run() which will iterate over all the 

610 statements in an ast.Module and for each ast.Assert statement it 

611 finds call .visit() with it. Then .visit_Assert() takes over and 

612 is responsible for creating new ast statements to replace the 

613 original assert statement: it rewrites the test of an assertion 

614 to provide intermediate values and replace it with an if statement 

615 which raises an assertion error with a detailed explanation in 

616 case the expression is false and calls pytest_assertion_pass hook 

617 if expression is true. 

618 

619 For this .visit_Assert() uses the visitor pattern to visit all the 

620 AST nodes of the ast.Assert.test field, each visit call returning 

621 an AST node and the corresponding explanation string. During this 

622 state is kept in several instance attributes: 

623 

624 :statements: All the AST statements which will replace the assert 

625 statement. 

626 

627 :variables: This is populated by .variable() with each variable 

628 used by the statements so that they can all be set to None at 

629 the end of the statements. 

630 

631 :variable_counter: Counter to create new unique variables needed 

632 by statements. Variables are created using .variable() and 

633 have the form of "@py_assert0". 

634 

635 :expl_stmts: The AST statements which will be executed to get 

636 data from the assertion. This is the code which will construct 

637 the detailed assertion message that is used in the AssertionError 

638 or for the pytest_assertion_pass hook. 

639 

640 :explanation_specifiers: A dict filled by .explanation_param() 

641 with %-formatting placeholders and their corresponding 

642 expressions to use in the building of an assertion message. 

643 This is used by .pop_format_context() to build a message. 

644 

645 :stack: A stack of the explanation_specifiers dicts maintained by 

646 .push_format_context() and .pop_format_context() which allows 

647 to build another %-formatted string while already building one. 

648 

649 This state is reset on every new assert statement visited and used 

650 by the other visitors. 

651 """ 

652

654 self, module_path: Optional[str], config: Optional[Config], source: bytes 

655 ) -> None: 

656 super().__init__() 

657 self.module_path = module_path 

658 self.config = config 

659 if config is not None: 

660 self.enable_assertion_pass_hook = config.getini( 

661 "enable_assertion_pass_hook"

663 else: 

664 self.enable_assertion_pass_hook = False 

665 self.source = source

671 def run(self, mod: ast.Module) -> None: 

672 """Find all assert statements in *mod* and rewrite them.""" 

673 if not mod.body: 

674 # Nothing to do. 

675 return 

676 

677 # We'll insert some special imports at the top of the module, but after any 

678 # docstrings and __future__ imports, so first figure out where that is. 

679 doc = getattr(mod, "docstring", None) 

680 expect_docstring = doc is None 

681 if doc is not None and self.is_rewrite_disabled(doc): 

682 return 

683 pos = 0 

684 lineno = 1 

685 for item in mod.body: 

686 if ( 

687 expect_docstring 

688 and isinstance(item, ast.Expr) 

689 and isinstance(item.value, ast.Str) 

690 ): 

691 doc = item.value.s 

692 if self.is_rewrite_disabled(doc): 

693 return 

694 expect_docstring = False 

695 elif ( 

696 isinstance(item, ast.ImportFrom) 

697 and item.level == 0 

698 and item.module == "__future__" 

699 ): 

700 pass 

701 else: 

702 break 

703 pos += 1 

704 # Special case: for a decorated function, set the lineno to that of the 

705 # first decorator, not the `def`. Issue #4984. 

706 if isinstance(item, ast.FunctionDef) and item.decorator_list: 

707 lineno = item.decorator_list[0].lineno 

708 else: 

709 lineno = item.lineno 

710 # Now actually insert the special imports. 

711 if sys.version_info >= (3, 10): 

712 aliases = [ 

713 ast.alias("builtins", "@py_builtins", lineno=lineno, col_offset=0), 

714 ast.alias( 

715 "_pytest.assertion.rewrite", 

716 "@pytest_ar", 

717 lineno=lineno, 

718 col_offset=0, 

719 ), 

720 ] 

721 else: 

722 aliases = [ 

723 ast.alias("builtins", "@py_builtins"), 

724 ast.alias("_pytest.assertion.rewrite", "@pytest_ar"), 

725 ] 

726 imports = [ 

727 ast.Import([alias], lineno=lineno, col_offset=0) for alias in aliases 

728 ] 

729 mod.body[pos:pos] = imports 

730 

731 # Collect asserts. 

732 nodes: List[ast.AST] = [mod] 

733 while nodes: 

734 node = nodes.pop() 

735 for name, field in ast.iter_fields(node): 

736 if isinstance(field, list): 

737 new: List[ast.AST] = [] 

738 for i, child in enumerate(field): 

739 if isinstance(child, ast.Assert): 

740 # Transform assert. 

741 new.extend(self.visit(child)) 

742 else: 

743 new.append(child) 

744 if isinstance(child, ast.AST): 

745 nodes.append(child) 

746 setattr(node, name, new) 

747 elif ( 

748 isinstance(field, ast.AST) 

749 # Don't recurse into expressions as they can't contain 

750 # asserts. 

751 and not isinstance(field, ast.expr) 

752 ): 

753 nodes.append(field)

787 def explanation_param(self, expr: ast.expr) -> str: 

788 """Return a new named %-formatting placeholder for expr. 

789 

790 This creates a %-formatting placeholder for expr in the 

791 current formatting context, e.g. ``%(py0)s``. The placeholder 

792 and expr are placed in the current format context so that it 

793 can be used on the next call to .pop_format_context(). 

794 """ 

795 specifier = "py" + str(next(self.variable_counter)) 

796 self.explanation_specifiers[specifier] = expr 

797 return "%(" + specifier + ")s"

799 def push_format_context(self) -> None: 

800 """Create a new formatting context. 

801 

802 The format context is used for when an explanation wants to 

803 have a variable value formatted in the assertion message. In 

804 this case the value required can be added using 

805 .explanation_param(). Finally .pop_format_context() is used 

806 to format a string of %-formatted values as added by 

807 .explanation_param(). 

808 """ 

809 self.explanation_specifiers: Dict[str, ast.expr] = {} 

810 self.stack.append(self.explanation_specifiers) 

811 

812 def pop_format_context(self, expl_expr: ast.expr) -> ast.Name: 

813 """Format the %-formatted string with current format context. 

814 

815 The expl_expr should be an str ast.expr instance constructed from 

816 the %-placeholders created by .explanation_param(). This will 

817 add the required code to format said string to .expl_stmts and 

818 return the ast.Name instance of the formatted string. 

819 """ 

820 current = self.stack.pop() 

821 if self.stack: 

822 self.explanation_specifiers = self.stack[-1] 

823 keys = [ast.Str(key) for key in current.keys()] 

824 format_dict = ast.Dict(keys, list(current.values())) 

825 form = ast.BinOp(expl_expr, ast.Mod(), format_dict) 

826 name = "@py_format" + str(next(self.variable_counter)) 

827 if self.enable_assertion_pass_hook: 

828 self.format_variables.append(name) 

829 self.expl_stmts.append(ast.Assign([ast.Name(name, ast.Store())], form)) 

830 return ast.Name(name, ast.Load())

833 """Handle expressions we don't have custom code for.""" 

834 assert isinstance(node, ast.expr) 

835 res = self.assign(node) 

836 return res, self.explanation_param(self.display(res))

838 def visit_Assert(self, assert_: ast.Assert) -> List[ast.stmt]: 

839 """Return the AST statements to replace the ast.Assert instance. 

840 

841 This rewrites the test of an assertion to provide 

842 intermediate values and replace it with an if statement which 

843 raises an assertion error with a detailed explanation in case 

844 the expression is false. 

845 """ 

846 if isinstance(assert_.test, ast.Tuple) and len(assert_.test.elts) >= 1: 

847 from _pytest.warning_types import PytestAssertRewriteWarning 

848 import warnings 

849 

850 # TODO: This assert should not be needed. 

851 assert self.module_path is not None 

852 warnings.warn_explicit( 

853 PytestAssertRewriteWarning( 

854 "assertion is always true, perhaps remove parentheses?" 

855 ), 

856 category=None, 

857 filename=os.fspath(self.module_path), 

858 lineno=assert_.lineno, 

859 ) 

860 

861 self.statements: List[ast.stmt] = [] 

862 self.variables: List[str] = [] 

863 self.variable_counter = itertools.count() 

864 

865 if self.enable_assertion_pass_hook: 

866 self.format_variables: List[str] = [] 

867 

868 self.stack: List[Dict[str, ast.expr]] = [] 

869 self.expl_stmts: List[ast.stmt] = [] 

870 self.push_format_context() 

871 # Rewrite assert into a bunch of statements. 

872 top_condition, explanation = self.visit(assert_.test) 

873 

874 negation = ast.UnaryOp(ast.Not(), top_condition) 

875 

876 if self.enable_assertion_pass_hook: # Experimental pytest_assertion_pass hook 

877 msg = self.pop_format_context(ast.Str(explanation)) 

878 

879 # Failed 

880 if assert_.msg: 

881 assertmsg = self.helper("_format_assertmsg", assert_.msg) 

882 gluestr = "\n>assert " 

883 else: 

884 assertmsg = ast.Str("") 

885 gluestr = "assert " 

886 err_explanation = ast.BinOp(ast.Str(gluestr), ast.Add(), msg) 

887 err_msg = ast.BinOp(assertmsg, ast.Add(), err_explanation) 

888 err_name = ast.Name("AssertionError", ast.Load()) 

889 fmt = self.helper("_format_explanation", err_msg) 

890 exc = ast.Call(err_name, [fmt], []) 

891 raise_ = ast.Raise(exc, None) 

892 statements_fail = [] 

893 statements_fail.extend(self.expl_stmts) 

894 statements_fail.append(raise_) 

895 

896 # Passed 

897 fmt_pass = self.helper("_format_explanation", msg) 

898 orig = self._assert_expr_to_lineno()[assert_.lineno] 

899 hook_call_pass = ast.Expr( 

900 self.helper( 

901 "_call_assertion_pass", 

902 ast.Num(assert_.lineno), 

903 ast.Str(orig), 

904 fmt_pass, 

905 ) 

906 ) 

907 # If any hooks implement assert_pass hook 

908 hook_impl_test = ast.If( 

909 self.helper("_check_if_assertion_pass_impl"), 

910 self.expl_stmts + [hook_call_pass], 

911 [], 

912 ) 

913 statements_pass = [hook_impl_test] 

914 

915 # Test for assertion condition 

916 main_test = ast.If(negation, statements_fail, statements_pass) 

917 self.statements.append(main_test) 

918 if self.format_variables: 

919 variables = [ 

920 ast.Name(name, ast.Store()) for name in self.format_variables 

921 ] 

922 clear_format = ast.Assign(variables, ast.NameConstant(None)) 

923 self.statements.append(clear_format) 

924 

925 else: # Original assertion rewriting 

926 # Create failure message. 

927 body = self.expl_stmts 

928 self.statements.append(ast.If(negation, body, [])) 

929 if assert_.msg: 

930 assertmsg = self.helper("_format_assertmsg", assert_.msg) 

931 explanation = "\n>assert " + explanation 

932 else: 

933 assertmsg = ast.Str("") 

934 explanation = "assert " + explanation 

935 template = ast.BinOp(assertmsg, ast.Add(), ast.Str(explanation)) 

936 msg = self.pop_format_context(template) 

937 fmt = self.helper("_format_explanation", msg) 

938 err_name = ast.Name("AssertionError", ast.Load()) 

939 exc = ast.Call(err_name, [fmt], []) 

940 raise_ = ast.Raise(exc, None) 

941 

942 body.append(raise_) 

943 

944 # Clear temporary variables by setting them to None. 

945 if self.variables: 

946 variables = [ast.Name(name, ast.Store()) for name in self.variables] 

947 clear = ast.Assign(variables, ast.NameConstant(None)) 

948 self.statements.append(clear) 

949 # Fix line numbers. 

950 for stmt in self.statements: 

951 set_location(stmt, assert_.lineno, assert_.col_offset) 

952 return self.statements

954 def visit_Name(self, name: ast.Name) -> Tuple[ast.Name, str]: 

955 # Display the repr of the name if it's a local variable or 

956 # _should_repr_global_name() thinks it's acceptable. 

957 locs = ast.Call(self.builtin("locals"), [], []) 

958 inlocs = ast.Compare(ast.Str(name.id), [ast.In()], [locs]) 

959 dorepr = self.helper("_should_repr_global_name", name) 

960 test = ast.BoolOp(ast.Or(), [inlocs, dorepr]) 

961 expr = ast.IfExp(test, self.display(name), ast.Str(name.id)) 

962 return name, self.explanation_param(expr)

964 def visit_BoolOp(self, boolop: ast.BoolOp) -> Tuple[ast.Name, str]: 

965 res_var = self.variable() 

966 expl_list = self.assign(ast.List([], ast.Load())) 

967 app = ast.Attribute(expl_list, "append", ast.Load()) 

968 is_or = int(isinstance(boolop.op, ast.Or)) 

969 body = save = self.statements 

970 fail_save = self.expl_stmts 

971 levels = len(boolop.values) - 1 

972 self.push_format_context() 

973 # Process each operand, short-circuiting if needed. 

974 for i, v in enumerate(boolop.values): 

975 if i: 

976 fail_inner: List[ast.stmt] = [] 

977 # cond is set in a prior loop iteration below 

978 self.expl_stmts.append(ast.If(cond, fail_inner, [])) # noqa 

979 self.expl_stmts = fail_inner 

980 self.push_format_context() 

981 res, expl = self.visit(v) 

982 body.append(ast.Assign([ast.Name(res_var, ast.Store())], res)) 

983 expl_format = self.pop_format_context(ast.Str(expl)) 

984 call = ast.Call(app, [expl_format], []) 

985 self.expl_stmts.append(ast.Expr(call)) 

986 if i < levels: 

987 cond: ast.expr = res 

988 if is_or: 

989 cond = ast.UnaryOp(ast.Not(), cond) 

990 inner: List[ast.stmt] = [] 

991 self.statements.append(ast.If(cond, inner, [])) 

992 self.statements = body = inner 

993 self.statements = save 

994 self.expl_stmts = fail_save 

995 expl_template = self.helper("_format_boolop", expl_list, ast.Num(is_or)) 

996 expl = self.pop_format_context(expl_template) 

997 return ast.Name(res_var, ast.Load()), self.explanation_param(expl)

999 def visit_UnaryOp(self, unary: ast.UnaryOp) -> Tuple[ast.Name, str]: 

1000 pattern = UNARY_MAP[unary.op.__class__] 

1001 operand_res, operand_expl = self.visit(unary.operand) 

1002 res = self.assign(ast.UnaryOp(unary.op, operand_res)) 

1003 return res, pattern % (operand_expl,)

1005 def visit_BinOp(self, binop: ast.BinOp) -> Tuple[ast.Name, str]: 

1006 symbol = BINOP_MAP[binop.op.__class__] 

1007 left_expr, left_expl = self.visit(binop.left) 

1008 right_expr, right_expl = self.visit(binop.right) 

1009 explanation = f"({left_expl} {symbol} {right_expl})" 

1010 res = self.assign(ast.BinOp(left_expr, binop.op, right_expr)) 

1011 return res, explanation

1013 def visit_Call(self, call: ast.Call) -> Tuple[ast.Name, str]: 

1014 new_func, func_expl = self.visit(call.func) 

1015 arg_expls = [] 

1016 new_args = [] 

1017 new_kwargs = [] 

1018 for arg in call.args: 

1019 res, expl = self.visit(arg) 

1020 arg_expls.append(expl) 

1021 new_args.append(res) 

1022 for keyword in call.keywords: 

1023 res, expl = self.visit(keyword.value) 

1024 new_kwargs.append(ast.keyword(keyword.arg, res)) 

1025 if keyword.arg: 

1026 arg_expls.append(keyword.arg + "=" + expl) 

1027 else: # **args have `arg` keywords with an .arg of None 

1028 arg_expls.append("**" + expl) 

1029 

1030 expl = "{}({})".format(func_expl, ", ".join(arg_expls)) 

1031 new_call = ast.Call(new_func, new_args, new_kwargs) 

1032 res = self.assign(new_call) 

1033 res_expl = self.explanation_param(self.display(res)) 

1034 outer_expl = f"{res_expl}\n{{{res_expl} = {expl}\n}}" 

1035 return res, outer_expl

1037 def visit_Starred(self, starred: ast.Starred) -> Tuple[ast.Starred, str]: 

1038 # A Starred node can appear in a function call. 

1039 res, expl = self.visit(starred.value) 

1040 new_starred = ast.Starred(res, starred.ctx) 

1041 return new_starred, "*" + expl

1043 def visit_Attribute(self, attr: ast.Attribute) -> Tuple[ast.Name, str]: 

1044 if not isinstance(attr.ctx, ast.Load): 

1045 return self.generic_visit(attr) 

1046 value, value_expl = self.visit(attr.value) 

1047 res = self.assign(ast.Attribute(value, attr.attr, ast.Load())) 

1048 res_expl = self.explanation_param(self.display(res)) 

1049 pat = "%s\n{%s = %s.%s\n}" 

1050 expl = pat % (res_expl, res_expl, value_expl, attr.attr) 

1051 return res, expl

1053 def visit_Compare(self, comp: ast.Compare) -> Tuple[ast.expr, str]: 

1054 self.push_format_context() 

1055 left_res, left_expl = self.visit(comp.left) 

1056 if isinstance(comp.left, (ast.Compare, ast.BoolOp)): 

1057 left_expl = f"({left_expl})" 

1058 res_variables = [self.variable() for i in range(len(comp.ops))] 

1059 load_names = [ast.Name(v, ast.Load()) for v in res_variables] 

1060 store_names = [ast.Name(v, ast.Store()) for v in res_variables] 

1061 it = zip(range(len(comp.ops)), comp.ops, comp.comparators) 

1062 expls = [] 

1063 syms = [] 

1064 results = [left_res] 

1065 for i, op, next_operand in it: 

1066 next_res, next_expl = self.visit(next_operand) 

1067 if isinstance(next_operand, (ast.Compare, ast.BoolOp)): 

1068 next_expl = f"({next_expl})" 

1069 results.append(next_res) 

1070 sym = BINOP_MAP[op.__class__] 

1071 syms.append(ast.Str(sym)) 

1072 expl = f"{left_expl} {sym} {next_expl}" 

1073 expls.append(ast.Str(expl)) 

1074 res_expr = ast.Compare(left_res, [op], [next_res]) 

1075 self.statements.append(ast.Assign([store_names[i]], res_expr)) 

1076 left_res, left_expl = next_res, next_expl 

1077 # Use pytest.assertion.util._reprcompare if that's available. 

1078 expl_call = self.helper( 

1079 "_call_reprcompare", 

1080 ast.Tuple(syms, ast.Load()), 

1081 ast.Tuple(load_names, ast.Load()), 

1082 ast.Tuple(expls, ast.Load()), 

1083 ast.Tuple(results, ast.Load()), 

1084 ) 

1085 if len(comp.ops) > 1: 

1086 res: ast.expr = ast.BoolOp(ast.And(), load_names) 

1087 else: 

1088 res = load_names[0] 

1089 return res, self.explanation_param(self.pop_format_context(expl_call))

1092def try_makedirs(cache_dir: Path) -> bool: 

1093 """Attempt to create the given directory and sub-directories exist. 

1094 

1095 Returns True if successful or if it already exists. 

1096 """ 

1097 try: 

1098 os.makedirs(os.fspath(cache_dir), exist_ok=True) 

1099 except (FileNotFoundError, NotADirectoryError, FileExistsError): 

1100 # One of the path components was not a directory: 

1101 # - we're in a zip file 

1102 # - it is a file 

1103 return False 

1104 except PermissionError: 

1105 return False 

1106 except OSError as e: 

1107 # as of now, EROFS doesn't have an equivalent OSError-subclass 

1108 if e.errno == errno.EROFS: 

1109 return False 

1110 raise 

1111 return True
