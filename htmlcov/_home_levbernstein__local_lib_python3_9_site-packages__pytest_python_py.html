700 def collect(self) -> Iterable[Union[nodes.Item, nodes.Collector]]: 

701 this_path = self.fspath.dirpath() 

702 init_module = this_path.join("__init__.py") 

703 if init_module.check(file=1) and path_matches_patterns( 

704 init_module, self.config.getini("python_files") 

705 ): 

706 yield Module.from_parent(self, fspath=init_module) 

707 pkg_prefixes: Set[py.path.local] = set() 

708 for direntry in visit(str(this_path), recurse=self._recurse): 

709 path = py.path.local(direntry.path) 

710 

711 # We will visit our own __init__.py file, in which case we skip it. 

712 if direntry.is_file(): 

713 if direntry.name == "__init__.py" and path.dirpath() == this_path: 

714 continue 

715 

716 parts_ = parts(direntry.path) 

717 if any( 

718 str(pkg_prefix) in parts_ and pkg_prefix.join("__init__.py") != path 

719 for pkg_prefix in pkg_prefixes 

720 ): 

721 continue 

722 

723 if direntry.is_file(): 

724 yield from self._collectfile(path) 

725 elif not direntry.is_dir(): 

726 # Broken symlink or invalid/missing file. 

727 continue 

728 elif path.join("__init__.py").check(file=1):

985 self,

988 indirect: Union[bool, Sequence[str]] = False,

997 _param_mark: Optional[Mark] = None, 

998 ) -> None: 

999 """Add new invocations to the underlying test function using the list 

1000 of argvalues for the given argnames. Parametrization is performed 

1001 during the collection phase. If you need to setup expensive resources 

1002 see about setting indirect to do it rather at test setup time. 

1003 

1004 :param argnames: 

1005 A comma-separated string denoting one or more argument names, or 

1006 a list/tuple of argument strings. 

1007 

1008 :param argvalues: 

1009 The list of argvalues determines how often a test is invoked with 

1010 different argument values. 

1011 

1012 If only one argname was specified argvalues is a list of values. 

1013 If N argnames were specified, argvalues must be a list of 

1014 N-tuples, where each tuple-element specifies a value for its 

1015 respective argname. 

1016 

1017 :param indirect: 

1018 A list of arguments' names (subset of argnames) or a boolean. 

1019 If True the list contains all names from the argnames. Each 

1020 argvalue corresponding to an argname in this list will

1023 setup phase of a test rather than at collection time. 

1024

1027 or a callable to return part of the id for each argvalue. 

1028

1044

1145 def _validate_ids( 

1146 self, 

1147 ids: Iterable[Union[None, str, float, int, bool]], 

1148 parameters: Sequence[ParameterSet], 

1149 func_name: str, 

1150 ) -> List[Union[None, str]]: 

1151 try: 

1152 num_ids = len(ids) # type: ignore[arg-type] 

1153 except TypeError: 

1154 try: 

1155 iter(ids) 

1156 except TypeError as e: 

1157 raise TypeError("ids must be a callable or an iterable") from e 

1158 num_ids = len(parameters) 

1159 

1160 # num_ids == 0 is a special case: https://github.com/pytest-dev/pytest/issues/1849 

1161 if num_ids != len(parameters) and num_ids != 0: 

1162 msg = "In {}: {} parameter sets specified, with different number of ids: {}" 

1163 fail(msg.format(func_name, len(parameters), num_ids), pytrace=False)

1165 new_ids = [] 

1166 for idx, id_value in enumerate(itertools.islice(ids, num_ids)): 

1167 if id_value is None or isinstance(id_value, str): 

1168 new_ids.append(id_value) 

1169 elif isinstance(id_value, (float, int, bool)): 

1170 new_ids.append(str(id_value)) 

1171 else: 

1172 msg = ( # type: ignore[unreachable] 

1173 "In {}: ids must be list of string/float/int/bool, " 

1174 "found: {} (type: {!r}) at index {}" 

1175 ) 

1176 fail( 

1177 msg.format(func_name, saferepr(id_value), type(id_value), idx), 

1178 pytrace=False, 

1179 ) 

1180 return new_ids
